# simulate returns
import numpy as np
import pandas as pd
%matplotlib inline
from matplotlib import pyplot as plt

returns = pd.DataFrame(np.random.normal(1.0, 0.3, (100, 10)))
prices = returns.cumprod()
prices.plot()
plt.title('Randomly-generated Prices')
plt.xlabel('Time')
plt.ylabel('Price')
plt.legend(loc=0);
